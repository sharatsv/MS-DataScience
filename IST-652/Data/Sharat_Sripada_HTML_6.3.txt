# urllib + beautifulsoup
from urllib import request
url = 'https://www.nytimes.com/2020/11/11/opinion/biden-trump-foreign-policy.html?action=click&module=Opinion&pgtype=Homepage'
response = request.urlopen(url)
html = response.read().decode('utf8')
html[:500]

# Using BeautifulSoup library
from bs4 import BeautifulSoup
urldata = BeautifulSoup(html, "html.parser")

# Search for anchors by finding all 'a' attribs 
anchors = urldata.find_all('a')
    
# Walk the list-type object anchors and find all the href instances
for i in anchors[-5:]:
    if i.has_attr('href'):
        print(i.prettify())
